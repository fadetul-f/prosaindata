import requests
from bs4 import BeautifulSoup
import csv
   
URL = "https://pta.trunojoyo.ac.id/"
page = requests.get(URL)
   
soup = BeautifulSoup(page.content, "html.parser")
results = soup.find(id="content_journal")
#print(results.prettify())

table = results.find_all(attrs={"data-id": True}) 
skripsi=[]  # a list to store skripsi
   
   
for row in table:
    data = {}
    data['penulis'] = row.find("span")
    data['judul'] = row.find("a", class_="title")
    data['abstrak'] = row.find("p")
    skripsi.append(data)
   
filename = 'pta.csv'
with open(filename, 'w', newline='') as f:
    w = csv.DictWriter(f,['penulis', 'judul', 'abstrak'])
    w.writeheader()
    for data in skripsi:
        w.writerow(data)

import requests
from bs4 import BeautifulSoup
import csv
   
URL = "https://www.liputan6.com/news"
page = requests.get(URL)
   
soup = BeautifulSoup(page.content, "html.parser")
results = soup.find(id="main")
#print(results.prettify())

table = results.find_all(attrs={"data-id": True}) 
berita=[]  # a list to store berita
   
   
for row in table:
    data = {}
    data['judul'] = row.find("h4", class_="articles--iridescent-list--text-item__header")
    data['isi'] = row.find("div", class_="articles--iridescent-list--text-item__summary articles--iridescent-list--text-item__summary-seo")
    berita.append(data)
   
filename = 'berita.csv'
with open(filename, 'w', newline='') as f:
    w = csv.DictWriter(f,['judul', 'isi'])
    w.writeheader()
    for data in berita:
        w.writerow(data)
